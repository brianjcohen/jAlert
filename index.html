Example Code (actual code from demo above):

<!-- Includes the CSS -->
<link rel="stylesheet" type="text/css" href="jAlert-v1.css">

<!-- Create buttons -->
<a href='#' class="btn btn-lg btn-success ajaxAlert" role="button">Ajax</a> 
<a href='#' class="btn btn-lg btn-success iframeAlert" role="button">iFrame</a> 
<a href='#' class="btn btn-lg btn-success photoAlert" role="button">Photo</a>
<a href='#' class="btn btn-lg btn-success staticAlert" role="button">Default</a>
<a href='#' class="btn btn-lg btn-success successAlert" role="button">Success</a>
<a href='#' class="btn btn-lg btn-success errorAlert" role="button">Error</a>

<!-- Includes the jQuery and the JS -->
<script src='https://code.jquery.com/jquery-2.0.3.min.js'></script>
<script src='jAlert-v1.js'></script>

<!-- Attach jConfirm to your buttons -->
<script>
$('body').on('click', '.ajaxAlert', function(){ 
ajaxAlert('Ajax Demo', 'ajax.php', 'sm', '', true, true, false); 
}); 
$('body').on('click', '.iframeAlert', function(){ 
iframeAlert('iFrame Demo', 'https://nomoreagent.com', '600px', 'md', '', true, true, false); 
}); 
$('body').on('click', '.photoAlert', function(){ 
showPic('http://farm4.staticflickr.com/3533/3872606628_9591e05d16_o.jpg', 'lg'); 
});
$('body').on('click', '.staticAlert', function(){ 
staticAlert('Successful Demo', 'You\'re in awe now aren\'t you?', 'sm', '', true, true, false); 
}); 
$('body').on('click', '.successAlert', function(){ 
staticAlert('Success! You did it!', 'So, now you know how to make your customer feel like they did a great job!', 'sm jSuccess', '', true, true, false); 
}); 
$('body').on('click', '.errorAlert', function(){ 
staticAlert('ERROR!', 'You better fix it soon or you\'ll lose your customer!', 'sm jError', '', true, true, false); 
});
</script>
